35double haversineDistance(double lat1, double lon1, double lat2, double lon2) {
36 double dLat = toRadians(lat2 - lat1);
37 double dLon = toRadians(lon2 - lon1);
38
39 double a = std::sin(dLat / 2) * std::sin(dLat / 2) +
40 std::cos(toRadians(lat1)) * std::cos(toRadians(lat2)) *
41 std::sin(dLon / 2) * std::sin(dLon / 2);
42
43 double c = 2 * std::atan2(std::sqrt(a), std::sqrt(1 - a));
44
45 return EARTH_RADIUS_KM * c;
46}

82std::vector<int> triangularApproximation(const std::vector<NodeData>& nodes, const std::vector<std::vector<double>>& distances) {
83 int numNodes = nodes.size();
84 std::vector<bool> visited(numNodes, false);
85 std::vector<int> tour;
86 tour.reserve(numNodes);
87
88 // Start at the node with the zero-identifier label
89 int currentNode = 0;
90 visited[currentNode] = true;
91 tour.push_back(currentNode);
92
93 // Select the nearest neighbor node not visited yet as the next node to visit
94 for (int i = 0; i < numNodes - 1; ++i) {
95 int nearestNeighbor = findNearestNeighbor(currentNode, visited, distances);
96 currentNode = nearestNeighbor;
97 visited[currentNode] = true;
98 tour.push_back(currentNode);
99 }
100
101 // Add an edge from the last visited node back to the starting node
102 tour.push_back(0);
103
104 return tour;
105}

129 distances[edge.destination][edge.origin] = edge.distance;
130 }
131
132 // Calculate the geographic distances for missing edges
133 for (int i = 0; i < numNodes; ++i) {
134 for (int j = i + 1; j < numNodes; ++j) {
135 if (distances[i][j] == 0.0) {
136 double lat1 = nodes[i].latitude;

142 distances[i][j] = distance;
143 distances[j][i] = distance;
144 }
145 }
146 }
