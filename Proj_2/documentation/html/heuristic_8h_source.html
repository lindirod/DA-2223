89 std::vector<std::vector<int>> performHierarchicalClustering(const std::vector<T>& data, int numClusters) {
90 int numPoints = data.size();
91
92 // Initialize each data point as a separate cluster
93 std::vector<std::vector<int>> clusters;
94 for (int i = 0; i < numPoints; ++i) {
95 clusters.push_back(std::vector<int>{i});
96 }
97
98 // Merge clusters until the desired number of clusters is reached
99 while (clusters.size() > numClusters) {
100 double minDistance = std::numeric_limits<double>::max();
101 int cluster1Index = -1;
102 int cluster2Index = -1;
103
104 // Find the closest pair of clusters
105 for (int i = 0; i < clusters.size(); ++i) {
106 for (int j = i + 1; j < clusters.size(); ++j) {
107 double distance = data[clusters[i][0]].distance;
108 if (distance < minDistance) {
109 minDistance = distance;
110 cluster1Index = i;
111 cluster2Index = j;
112 }
113 }
114 }
115
116 // Merge the closest pair of clusters
117 clusters[cluster1Index].insert(clusters[cluster1Index].end(), clusters[cluster2Index].begin(), clusters[cluster2Index].end());
118 clusters.erase(clusters.begin() + cluster2Index);
119 }
120
121 return clusters;
122 }

136 std::vector<int> performDivideAndConquer(const std::vector<std::vector<int>>& clusters, const std::vector<std::vector<double>>& distances) {
